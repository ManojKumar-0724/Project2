PSEUDO CODES FOR HERITAGE MONUMENTS WEB APPLICATION
====================================================

1. USER AUTHENTICATION MODULE
==============================

ALGORITHM UserSignUp
INPUT: email, password, name
OUTPUT: User account created and authenticated

BEGIN
  VALIDATE email format
  VALIDATE password strength (min 6 characters)
  
  TRY
    CALL Supabase.auth.signUp(email, password)
    IF signup successful THEN
      CREATE profile record with user metadata
      SET role = 'user' in user_roles table
      REDIRECT to home page
    END IF
  CATCH error
    DISPLAY error message to user
  END TRY
END

ALGORITHM UserSignIn
INPUT: email, password
OUTPUT: Authenticated session

BEGIN
  VALIDATE input fields not empty
  
  TRY
    CALL Supabase.auth.signInWithPassword(email, password)
    IF login successful THEN
      FETCH user role from user_roles table
      STORE session in local storage
      REDIRECT to home page
    END IF
  CATCH error
    DISPLAY "Invalid credentials" message
  END TRY
END

ALGORITHM CheckUserRole
OUTPUT: User role (admin/moderator/user)

BEGIN
  GET current user ID from session
  IF user not authenticated THEN
    RETURN null
  END IF
  
  QUERY user_roles WHERE user_id = current_user_id
  IF role exists THEN
    RETURN role
  ELSE
    RETURN 'user' (default)
  END IF
END


2. MONUMENT BROWSING AND SEARCH MODULE
=======================================

ALGORITHM BrowseMonuments
INPUT: search_query (optional), filters (optional)
OUTPUT: List of monuments

BEGIN
  INITIALIZE monuments_list = []
  
  IF search_query is empty THEN
    QUERY all monuments FROM database
      ORDER BY name ASC
  ELSE
    QUERY monuments WHERE 
      name CONTAINS search_query OR
      location CONTAINS search_query OR
      description CONTAINS search_query
  END IF
  
  IF filters applied THEN
    FILTER monuments by location/era/type
  END IF
  
  FOR each monument IN monuments_list
    FETCH monument image from storage
    CALCULATE average rating from ratings table
    CHECK if monument is in user favorites
  END FOR
  
  RETURN monuments_list
END

ALGORITHM GetMonumentDetails
INPUT: monument_id
OUTPUT: Complete monument information

BEGIN
  QUERY monument WHERE id = monument_id
  IF monument not found THEN
    DISPLAY "Monument not found"
    RETURN null
  END IF
  
  FETCH related stories from stories table
  FETCH quiz templates for monument
  CALCULATE total views from monument_views table
  GET user rating for this monument
  
  INCREMENT view count
  INSERT into monument_views table
  
  RETURN monument with all related data
END


3. STORY MANAGEMENT MODULE
===========================

ALGORITHM DisplayStories
INPUT: monument_id, story_length (short/medium/detailed)
OUTPUT: Story content with audio narration

BEGIN
  QUERY stories WHERE monument_id = monument_id 
    AND status = 'approved'
  
  FOR each story IN stories_list
    IF story_length = 'short' THEN
      content = TRUNCATE(story.content, 75 words)
    ELSE IF story_length = 'medium' THEN
      content = TRUNCATE(story.content, 150 words)
    ELSE
      content = story.content (full)
    END IF
    
    DISPLAY content with audio button
  END FOR
END

ALGORITHM GenerateAudioNarration
INPUT: text, language (English/Kannada)
OUTPUT: Audio playback

BEGIN
  INITIALIZE speech_synthesis = SpeechSynthesis API
  
  IF language = 'English' THEN
    SET voice_code = 'en-US'
  ELSE IF language = 'Kannada' THEN
    SET voice_code = 'kn-IN'
  END IF
  
  GET available voices for language
  IF voices available THEN
    CREATE utterance with text and voice
    SET speech rate and pitch
    PLAY audio
  ELSE
    DISPLAY "Voice not available" error
  END IF
END

ALGORITHM ContributeStory
INPUT: monument_id, title, content, language
OUTPUT: Story submitted for moderation

BEGIN
  GET current user from session
  IF user not authenticated THEN
    REDIRECT to login page
  END IF
  
  VALIDATE title not empty
  VALIDATE content length > 100 characters
  
  INSERT INTO stories TABLE
    monument_id = monument_id
    title = title
    content = content
    author_id = current_user_id
    language = language
    status = 'pending'
    created_at = current_timestamp
  
  DISPLAY "Story submitted for review" message
  REDIRECT to stories page
END


4. QUIZ SYSTEM MODULE
======================

ALGORITHM LoadQuiz
INPUT: monument_id
OUTPUT: Quiz questions with options

BEGIN
  QUERY quiz_templates WHERE monument_id = monument_id
    AND is_active = true
  
  IF quiz_template exists THEN
    QUERY quiz_questions WHERE quiz_template_id = template_id
      ORDER BY order_index
    
    IF questions exist THEN
      RETURN stored questions
    END IF
  END IF
  
  // Fallback to AI generation
  CALL GenerateQuizWithAI(monument_id)
  RETURN generated questions
END

ALGORITHM ProcessQuizAnswers
INPUT: quiz_id, user_answers[]
OUTPUT: Score and feedback

BEGIN
  INITIALIZE correct_count = 0
  INITIALIZE total_questions = length(user_answers)
  
  FOR each answer IN user_answers
    GET correct_answer from quiz_questions
    IF answer = correct_answer THEN
      correct_count = correct_count + 1
    END IF
  END FOR
  
  CALCULATE score = (correct_count / total_questions) * 100
  
  INSERT INTO quiz_completions TABLE
    user_id = current_user_id
    quiz_template_id = quiz_id
    score = score
    completed_at = current_timestamp
  
  RETURN score with explanations
END

ALGORITHM GenerateQuizWithAI
INPUT: monument_id
OUTPUT: AI-generated quiz questions

BEGIN
  GET monument details from database
  
  CREATE prompt = "Generate 5 multiple-choice questions about " + monument.name
  
  CALL Edge Function generate-quiz
    WITH monument context
  
  IF generation successful THEN
    PARSE JSON response
    VALIDATE question format
    RETURN questions array
  ELSE
    THROW "Quiz generation failed" error
  END IF
END


5. ADMIN PANEL MODULE
======================

ALGORITHM AccessAdminPanel
OUTPUT: Admin dashboard or access denied

BEGIN
  GET current user from session
  IF user not authenticated THEN
    DISPLAY "Please sign in" message
    RETURN
  END IF
  
  GET user_role = CheckUserRole()
  IF user_role != 'admin' AND user_role != 'moderator' THEN
    DISPLAY "Access denied - Admin privileges required"
    RETURN
  END IF
  
  LOAD admin dashboard components
END

ALGORITHM ModerateStories
OUTPUT: List of pending stories

BEGIN
  QUERY stories WHERE status = 'pending'
    ORDER BY created_at DESC
  
  FOR each story IN pending_stories
    DISPLAY story details
    SHOW approve/reject buttons
  END FOR
END

ALGORITHM ApproveStory
INPUT: story_id
OUTPUT: Story approved and published

BEGIN
  VALIDATE user has admin/moderator role
  
  UPDATE stories TABLE
    SET status = 'approved'
    WHERE id = story_id
  
  // Optional: Send notification to author
  NOTIFY author "Your story has been approved"
  
  DISPLAY "Story approved successfully"
  REFRESH pending stories list
END

ALGORITHM RejectStory
INPUT: story_id, rejection_reason
OUTPUT: Story rejected with feedback

BEGIN
  VALIDATE user has admin/moderator role
  VALIDATE rejection_reason not empty
  
  UPDATE stories TABLE
    SET status = 'rejected'
    SET rejection_reason = rejection_reason
    WHERE id = story_id
  
  // Optional: Send notification to author
  NOTIFY author "Story rejected: " + rejection_reason
  
  DISPLAY "Story rejected"
  REFRESH pending stories list
END

ALGORITHM CreateStoryAsAdmin
INPUT: monument_id, title, content, author_name, language
OUTPUT: Story created and published

BEGIN
  VALIDATE user has admin role
  VALIDATE all required fields
  
  INSERT INTO stories TABLE
    monument_id = monument_id
    title = title
    content = content
    author = author_name
    language = language
    status = 'approved'  // Auto-approve admin stories
    created_by = current_user_id
    created_at = current_timestamp
  
  DISPLAY "Story created successfully"
END


6. FAVORITES MANAGEMENT MODULE
===============================

ALGORITHM ToggleFavorite
INPUT: monument_id
OUTPUT: Favorite added or removed

BEGIN
  GET current_user_id from session
  
  CHECK IF favorite exists WHERE
    user_id = current_user_id AND
    monument_id = monument_id
  
  IF favorite exists THEN
    DELETE FROM favorites
    DISPLAY "Removed from favorites"
  ELSE
    INSERT INTO favorites (user_id, monument_id, created_at)
    DISPLAY "Added to favorites"
  END IF
  
  UPDATE UI to reflect change
END

ALGORITHM GetUserFavorites
OUTPUT: List of favorite monuments

BEGIN
  GET current_user_id from session
  
  QUERY favorites JOIN monuments
    WHERE user_id = current_user_id
    ORDER BY created_at DESC
  
  FOR each favorite IN favorites_list
    FETCH monument details
    FETCH monument image
    CALCULATE rating
  END FOR
  
  RETURN favorites_list
END


7. RATING SYSTEM MODULE
========================

ALGORITHM RateMonument
INPUT: monument_id, rating_value (1-5)
OUTPUT: Rating saved

BEGIN
  GET current_user_id from session
  VALIDATE rating_value between 1 and 5
  
  CHECK IF existing rating WHERE
    user_id = current_user_id AND
    monument_id = monument_id
  
  IF existing rating THEN
    UPDATE ratings SET rating = rating_value
  ELSE
    INSERT INTO ratings (user_id, monument_id, rating, created_at)
  END IF
  
  CALCULATE new average rating for monument
  DISPLAY "Thank you for rating"
END

ALGORITHM GetAverageRating
INPUT: monument_id
OUTPUT: Average rating and count

BEGIN
  QUERY ratings WHERE monument_id = monument_id
  
  IF ratings exist THEN
    CALCULATE average = SUM(rating) / COUNT(ratings)
    CALCULATE total_ratings = COUNT(ratings)
    RETURN (average, total_ratings)
  ELSE
    RETURN (0, 0)
  END IF
END


8. AR VIEWER MODULE
====================

ALGORITHM InitializeARViewer
INPUT: monument_id
OUTPUT: AR scene loaded

BEGIN
  GET monument details from database
  GET 3D model URL from storage
  
  INITIALIZE A-Frame scene
  CREATE camera entity
  CREATE lighting entities
  
  LOAD 3D model (GLTF format)
    WITH position = (0, 0, -5)
    WITH scale = (1, 1, 1)
    WITH rotation enabled
  
  ADD environment lighting
  ADD orbit controls for interaction
  
  DISPLAY AR scene
  SHOW instructions "Drag to rotate, pinch to zoom"
END

ALGORITHM RotateModel
INPUT: touch_delta_x, touch_delta_y
OUTPUT: Model rotated

BEGIN
  GET current model rotation
  
  CALCULATE new_rotation_y = current_rotation.y + touch_delta_x * sensitivity
  CALCULATE new_rotation_x = current_rotation.x + touch_delta_y * sensitivity
  
  CLAMP rotation_x between -90 and 90 degrees
  
  UPDATE model rotation
  RENDER scene
END


9. ANALYTICS MODULE
====================

ALGORITHM TrackMonumentView
INPUT: monument_id
OUTPUT: View recorded

BEGIN
  GET current_user_id from session (if authenticated)
  GET session_id from browser
  
  INSERT INTO monument_views TABLE
    monument_id = monument_id
    user_id = current_user_id (nullable)
    session_id = session_id
    viewed_at = current_timestamp
    
  UPDATE monuments SET view_count = view_count + 1
    WHERE id = monument_id
END

ALGORITHM GetAnalytics
OUTPUT: Dashboard analytics data

BEGIN
  VALIDATE user has admin role
  
  CALCULATE total_monuments = COUNT(monuments)
  CALCULATE total_stories = COUNT(stories WHERE status = 'approved')
  CALCULATE total_users = COUNT(profiles)
  CALCULATE total_quiz_completions = COUNT(quiz_completions)
  
  GET top_rated_monuments = QUERY monuments
    ORDER BY average_rating DESC
    LIMIT 5
  
  GET most_viewed_monuments = QUERY monuments
    ORDER BY view_count DESC
    LIMIT 5
  
  GET recent_activities = QUERY monument_views
    ORDER BY viewed_at DESC
    LIMIT 10
  
  RETURN analytics_dashboard_data
END


10. SEARCH AND FILTER MODULE
=============================

ALGORITHM SearchMonuments
INPUT: search_term, filters
OUTPUT: Filtered monuments list

BEGIN
  INITIALIZE query = "SELECT * FROM monuments WHERE 1=1"
  
  IF search_term not empty THEN
    query += " AND (name ILIKE '%' || search_term || '%'"
    query += " OR location ILIKE '%' || search_term || '%'"
    query += " OR description ILIKE '%' || search_term || '%')"
  END IF
  
  IF filters.location specified THEN
    query += " AND location = " + filters.location
  END IF
  
  IF filters.era specified THEN
    query += " AND era = " + filters.era
  END IF
  
  IF filters.min_rating specified THEN
    query += " AND average_rating >= " + filters.min_rating
  END IF
  
  EXECUTE query
  RETURN results
END
